def show_error(key)
  if errors.has_key?(key)
    m = ''
    err = errors[key].first
    if key == :email and err == 'shinfo.blank.email'
      m = I18n.t(err, :value => 5)
      
    else
      m = I18n.t(err)
    end
    content_tag(:span, m, :class => 'form_error')
  end
end

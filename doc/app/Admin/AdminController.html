def create
  user = User.authenticate(params[:username], params[:password])
  if user.present?
    session[:user_id] = user
    redirect_to admin_product_path, :notice => t('login.login_success')
    
  else
    flash.now[:alert] = t('login.error')
    render :action => 'new'
  end
end
